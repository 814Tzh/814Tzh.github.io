# 省市区级联选择

vue2版本（el-cascader、递归、解构、计算属性）

<el-form-item label="省市区地址" prop="provinceName">
  <el-cascader
    ref="areaRef"
    :props="areaProps"
    :placeholder="addressPlaceholder"
    size="medium"
    clearable
    @change="handleAreaChange"
  />
</el-form-item>

data() {
    return {
	 // 地区懒加载
      areaProps: this.handleAreaProps()
  	}
 },

//处理回显的一种方案，利用computed显示在placeholder上
  computed: {
    addressPlaceholder() {
      const { provinceName, cityName, areaName } = this.form
      return provinceName ? `${provinceName}/${cityName}/${areaName}` : '请选择地址'
    }
  },

// 处理地区选项懒加载 先获取省级列表在根据id递归获取市级和区级列表
handleAreaProps() {
  return {
    lazy: true,
    async lazyLoad(node, resolve) {
      // // console.log('node', node)
      const { level, value = {}} = node
      // const reqFn = () => (level !== 0 ? getListById(value.id) : getListByLevel())
      const { data } = level !== 0 ? await getListById(value.id) : await getListByLevel()
      const nodes = data.list.map((item) => ({
        value: item,
        label: item.name,
        leaf: level >= 2
      }))
      resolve(nodes)
    }
  }
},

// 省市区code和name赋值
handleAreaChange(e) {
  console.log('handleAreaChange', e)
  if (!e || !e.length) {
    e = new Array(3).fill({})
  }
  const [{ name: provinceName, id: provinceCode }, { name: cityName, id: cityCode }, 
		{ name:areaName, id: areaCode }] = e
  this.form = {
    ...this.form,
    ...{ provinceCode, cityCode, areaCode, provinceName, cityName, areaName }
  }
}

# 文件上传组件

vue3版本（el-upload）

可自定义文件数量、文件大小限制、文件类型、提示文字显示隐藏

<template>
  <div class="upload-file">
    <el-upload
      ref="fileUpload"
      v-model:file-list="fileList"
      :show-file-list="false"
      :action="uploadFileUrl"
      :before-upload="handleBeforeUpload"
      :limit="limit"
      :on-error="handleUploadError"
      :on-exceed="handleExceed"
      :on-success="handleUploadSuccess"
      :headers="headers"
      multiple
      class="upload-file-uploader"
    >
      <!-- 上传按钮 -->
      <el-button type="primary">选取文件</el-button>
    </el-upload>
    <!-- 上传提示 -->
    <div v-if="showTip" class="el-upload__tip">
      请上传
      <template v-if="fileSize"> 大小不超过 <b style="color: #f56c6c">{{ fileSize }}MB</b> </template>
      <template v-if="fileType"> 格式为 <b style="color: #f56c6c">{{ fileType.join("/") }}</b> 				</template>
      的文件
    </div>
    <!-- 文件列表 -->
    <div class="file-list">
      <div v-for="(file, index) in fileList" :key="file.uid" class="file-list-item">
        <el-link :href="file.url" :underline="false" target="_blank">
          <span class="el-icon-document"> {{ file.name }} </span>
        </el-link>
        <el-link :underline="false" type="danger" @click="handleDelete(index)">删除</el-link>
      </div>
    </div>
  </div>
</template>

<script setup name="FileUpload">
import { getToken } from '@/utils/auth'

//参考withDefaults()重新配置props
const props = defineProps({
  modelValue: [String, Object, Array],
  // 数量限制
  limit: {
    type: Number,
    default: 5,
  },
  // 大小限制(MB)
  fileSize: {
    type: Number,
    default: 5,
  },
  // 文件类型, 例如['png', 'jpg', 'jpeg']
  fileType: {
    type: Array,
    default: () => ['doc', 'xls', 'ppt', 'pdf', 'png', 'jpg'],
  },
  // 是否显示提示
  isShowTip: {
    type: Boolean,
    default: true
  }
})

const emit = defineEmits(['modelValue'])

// 调用全局方法
const { proxy } = getCurrentInstance()

// 上传地址和上传头配置（详解看下面）
const link = import.meta.env.VITE_APP_ENV === 'development' ? '/zjwl' : 'https://test.com/zjwl'
const uploadFileUrl = ref(link + '/oss/uploadFile') // 上传文件服务器地址
const headers = ref({ Authorization: 'Bearer ' + getToken() })

// 提示是否展示
const showTip = computed(
  () => props.isShowTip && (props.fileType || props.fileSize)
)

//上传文件列表
const uploadList = ref([])
//文件列表
const fileList = ref([])

/* -------------------回显处理--------------------------- */
watch(() => props.modelValue, val => {
  if (val) {
    let temp = 1
    // 首先将值转为数组
    const list = Array.isArray(val) ? val : props.modelValue.split(',')
    // 然后将数组转为对象数组
    fileList.value = list.map(item => {
      if (typeof item === 'string') {
        item = { name: item, url: item }
      }
      //生成随机uid
      item.uid = item.uid || new Date().getTime() + temp++
      return item
    })
  } else {
    fileList.value = []
    return []
  }
}, { deep: true, immediate: true })

/* -------------------上传前校检格式和大小--------------------------- */
// 单次文件上传数量
const number = ref(0)

const handleBeforeUpload = (file) => {
  // 校检文件类型
  if (props.fileType.length) {
    let fileExtension = ''
    if (file.name.lastIndexOf('.') > -1) {
      fileExtension = file.name.slice(file.name.lastIndexOf('.') + 1)
    }
    const isTypeOk = props.fileType.some((type) => {
      if (file.type.indexOf(type) > -1) return true
      if (fileExtension && fileExtension.indexOf(type) > -1) return true
      return false
    })
    if (!isTypeOk) {
      proxy.$modal.msgError(`文件格式不正确, 请上传${props.fileType.join('/')}格式文件!`)
      return false
    }
  }
  // 校检文件大小
  if (props.fileSize) {
    const isLt = file.size / 1024 / 1024 < props.fileSize
    if (!isLt) {
      proxy.$modal.msgError(`上传文件大小不能超过 ${props.fileSize} MB!`)
      return false
    }
  }
  proxy.$modal.loading('正在上传文件，请稍候...')
  number.value++
  return true
}

/* -------------------上传后处理--------------------------- */
// 上传成功回调
function handleUploadSuccess(res, file) {
  console.log('file', file)
  if (res.respCode === '200') {
    uploadList.value.push({ name: file.name, url: res.data })
    console.log('uploadList.value', uploadList.value)
    uploadedSuccessfully()
  } else {
    number.value--
    proxy.$modal.closeLoading()
    proxy.$modal.msgError(res.respMsg)
    // element提供手动移除文件方法
    proxy.$refs.fileUpload.handleRemove(file)
    uploadedSuccessfully()
  }
}

// 上传结束处理
function uploadedSuccessfully() {
  if (number.value > 0 && uploadList.value.length === number.value) {
    fileList.value = fileList.value.filter(f => f.url !== undefined).concat(uploadList.value)
    uploadList.value = []
    number.value = 0
    emit('update:modelValue', fileList.value)
    proxy.$modal.closeLoading()
  }
}

// 上传失败
const handleUploadError = () => {
  proxy.$modal.msgError('上传文件失败')
}

// 文件个数超出
function handleExceed() {
  proxy.$modal.msgError(`上传文件数量不能超过 ${props.limit} 个!`)
}

// 删除文件
function handleDelete(index) {
  fileList.value.splice(index, 1)
  emit('update:modelValue', fileList.value)
}
</script>

上传地址和上传头配置详解

在开发环境中存在跨域问题，所以根据所属环境要进行不同的配置

//vite.config.ts
//用配置代理服务器处理跨域

    // vite 相关配置
    server: {
      port: 80,
      host: true,
      open: true,
      proxy: {
        '/zjwl': {
          target: 'https://test.com/zjwl',
          changeOrigin: true,
          //匹配到'/zjwl'时会与target进行拼接为'https://test.com/zjwl/zjwl'，所以需要去除一个/zjwl
          rewrite: (p) => p.replace(/^\/zjwl/, '')
        },
      }
    },

//如组件中代码
//假设当前处于开发环境取得link='/zjwl'
const link = import.meta.env.VITE_APP_ENV === 'development' ? '/zjwl' : 'https://test.com/zjwl'
//然后根据vite配置，uploadFileUrl最终地址为'https://test.com/zjwl/oss/uploadFile'
const uploadFileUrl = ref(link + '/oss/uploadFile') // 上传文件服务器地址
