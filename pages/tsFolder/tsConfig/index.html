# 基础配置

{
//tsconfig.json是ts编译器的配置文件，ts编译器
//路径：**表示任意目录 *表示任意文件
  // 需编译文件列表
  "files": [
    "./src/index.ts"
  ],
  // 包含匹配器
  "include": [
    "src/**/*"
  ],
  // 排除匹配器
  "exclude": [
    "node_modules",
    "**/*.js"
  ],
  // 配置继承
  "extends": "./ts/base",
  // 保存时编译
  "compileOnSave": {},
  // 编译配置
  "compilerOptions": {},
  // 工程引用
  "references": []
}
